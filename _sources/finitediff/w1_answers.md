```{code-cell} ipython3
# Heun's method
def my_heun(f,t,x0):

    nstep = len(t)
    h = t[1] - t[0]
    x = np.zeros((len(x0),nstep))
    x[:,0] = x0

    for k in range(nstep-1):
        s1 = f(t[k], x[:,k])
        s2 = f(t[k+1], x[:,k]+[h*a for a in s1])
        x[:,k+1] = x[:,k]+[h/2*a for a in np.add(s1,s2)]

    return x

# Euler's implicit method
def my_euli(f,t,X0):

    n = len(t)
    h = t[1]-t[0]
    X = np.zeros((len(X0),n))
    X[:,0] = X0

    for k in range(n-1):
        Z0 = X[:,k]
        X[:,k+1] = fsolve(lambda Z1: Z1-Z0-[h*a for a in f(t[k], Z1)], Z0)

    return X

# Euler's explicit method
def my_eulr(f,x,y0):

    nstep = len(x)
    h = x[1]-x[0]
    y = np.zeros((len(y0),nstep))
    y[:,0]=y0

    for k in range(nstep-1):
        y[:,k+1] = y[:,k]+[h*a for a in f(x[k], y[:,k])]
    return y
```

```{code-cell} ipython3
# Set up grids
ht = 0.1
t = np.arange(0,100,ht)
N = len(t)

x = np.zeros((1,N))[0]
x[0] = 1
v = np.zeros((1,N+1))[0]
v[0] = 0

#Leap frog method
for i in range(N-1):
    v[i+1] = v[i]-x[i]*ht
    x[i+1] = x[i]+v[i+1]*ht

v[N] = v[N-1]-x[N-1]*ht
v = (v[:-1]+v[1:])/2

# Compute differenced from extact energy result
plt.plot(t,abs(v**2+x**2-1)/2,'r')
plt.show()
```
